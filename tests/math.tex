\begin{algorithm}[H]
    \def\smallSkip{3pt}
    \def\medSkip{7pt}
    \setstretch{1.17}

    \KwData{Menge an $n > 3$ Punkten $P = [ (x_1, y_1), (x_2, y_2), \dots, (x_n, y_n) ]$}
    \KwResult{Kosten $m$ einer günstigsten Plottertour}\vspace{\medSkip}
    $A[1 \ldots n,1 \dots n] \gets $ $n \times n$ array\;
    $A[1,2] \gets \delta(P[1],P[2])$\label{alg:opt triv start}\;%
    \For{$j \gets 3, \dots, n$}{
        \For{$i \gets 1, \dots, j-2$}{\label{alg:opt opt i+1 start}
            $A[i,j] \gets A[i,j-1] + \delta(P[j-1], P[j])$\;
        }\label{alg:opt opt i+1 end}
        $A[j-1,j] \gets \min_{k \in \set{1, \dots, j-2}} A[k,j-1] + \delta(P[k],P[j])$\label{alg: first min}\label{alg:inner min}\;
    }
    \Return $\min_{k \in \set{1, \dots, n-1}} A[k,n] + \delta(P[k],P[n])$\label{alg:end all}\;
    \vspace{\medSkip}
    \caption{Berechnet Kosten einer günstigsten Plottertour}
\end{algorithm}
